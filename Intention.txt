CREATE TABLE Organisme(
   SIRET NUMBER(14),
   raison_Sociale VARCHAR2(50),
   PRIMARY KEY(SIRET)
);

CREATE TABLE territoire(
   IdT NUMBER(10),
   nom_territoire VARCHAR2(50),
   PRIMARY KEY(IdT)
);

CREATE TABLE grade(
   IdGr NUMBER(10),
   PRIMARY KEY(IdGr)
);

CREATE TABLE Titre(
   IdTi NUMBER(10),
   PRIMARY KEY(IdTi)
);

CREATE TABLE Dignite(
   IdD NUMBER(10),
   PRIMARY KEY(IdD)
);

CREATE TABLE rang(
   IdRa NUMBER(10),
   PRIMARY KEY(IdRa)
);

CREATE TABLE composant(
   IdC NUMBER(10),
   type_aliment VARCHAR2(50),
   nom VARCHAR2(50),
   allergene VARCHAR2(50),
   PRIMARY KEY(IdC)
);

CREATE TABLE Legume(
   IdL NUMBER(10),
   Verifie NUMBER(1),
   nom VARCHAR2(50),
   PRIMARY KEY(IdL)
);

CREATE TABLE machine(
   IdM NUMBER(10),
   nom VARCHAR2(50),
   PRIMARY KEY(IdM)
);

CREATE TABLE modele(
   IdMo NUMBER(10),
   nom_modele VARCHAR2(50),
   PRIMARY KEY(IdMo)
);

CREATE TABLE date_(
   date_val VARCHAR2(50),
   PRIMARY KEY(date_val)
);

CREATE TABLE Repas(
   IdR NUMBER(10),
   nom_repas VARCHAR2(50) NOT NULL,
   Date_R DATE NOT NULL,
   adr_repas VARCHAR2(50),
   Nom_Chevalier_Dame VARCHAR2(50) NOT NULL,
   PRIMARY KEY(IdR)
);

CREATE TABLE Sauce(
   IdC NUMBER(10),
   PRIMARY KEY(IdC),
   FOREIGN KEY(IdC) REFERENCES composant(IdC)
);

CREATE TABLE Ingredient(
   IdC NUMBER(10),
   PRIMARY KEY(IdC),
   FOREIGN KEY(IdC) REFERENCES composant(IdC)
);

CREATE TABLE Membre(
   CodeMembre VARCHAR2(50),
   nom_Membre VARCHAR2(255) NOT NULL,
   adresse VARCHAR2(50) NOT NULL,
   courriel VARCHAR2(50) NOT NULL,
   num_tel VARCHAR2(50) NOT NULL,
   IdD NUMBER(10),
   IdTi NUMBER(10),
   IdRa NUMBER(10),
   IdGr NUMBER(10) NOT NULL,
   PRIMARY KEY(CodeMembre),
   FOREIGN KEY(IdD) REFERENCES Dignite(IdD),
   FOREIGN KEY(IdTi) REFERENCES Titre(IdTi),
   FOREIGN KEY(IdRa) REFERENCES rang(IdRa),
   FOREIGN KEY(IdGr) REFERENCES grade(IdGr)
);

CREATE TABLE organisation(
   IdO NUMBER(10),
   nom_orga VARCHAR2(50) NOT NULL,
   type_orga VARCHAR2(50) NOT NULL,
   IdT NUMBER(10) NOT NULL,
   PRIMARY KEY(IdO),
   FOREIGN KEY(IdT) REFERENCES territoire(IdT)
);

CREATE TABLE Groupe(
   IdG NUMBER(10),
   IdR NUMBER(10) NOT NULL,
   PRIMARY KEY(IdG),
   UNIQUE(IdR),
   FOREIGN KEY(IdR) REFERENCES Repas(IdR)
);

CREATE TABLE plat(
   IdP NUMBER(10),
   IdL NUMBER(10),
   PRIMARY KEY(IdP),
   FOREIGN KEY(IdL) REFERENCES Legume(IdL)
);

CREATE TABLE Entretien(
   IdE NUMBER(10),
   Certificat_d_entretien TIMESTAMP,
   CodeMembre VARCHAR2(50) NOT NULL,
   PRIMARY KEY(IdE),
   FOREIGN KEY(CodeMembre) REFERENCES Membre(CodeMembre)
);

CREATE TABLE Ordre(
   IdO NUMBER(10),
   PRIMARY KEY(IdO),
   FOREIGN KEY(IdO) REFERENCES organisation(IdO)
);

CREATE TABLE Club(
   IdO_1 NUMBER(10),
   IdO NUMBER(10) NOT NULL,
   PRIMARY KEY(IdO_1),
   FOREIGN KEY(IdO_1) REFERENCES organisation(IdO),
   FOREIGN KEY(IdO) REFERENCES Ordre(IdO)
);
